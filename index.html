  1. SELECT * FROM `students` WHERE `date_of_birth` >= '1990-01-01' AND `date_of_birth` <= '1990-12-31';
  2. SELECT * FROM `courses` WHERE `cfu` > 10;
  3. SELECT * FROM `students` WHERE `date_of_birth` < CONCAT(YEAR(CURRENT_DATE)-30,'-01-01');
  4. SELECT * FROM `courses` WHERE `period` = 'I semestre' AND `year` = 1;
  5. SELECT * FROM `exams` WHERE `date` = '2020-06-20' AND `hour` >= '14:00:00';
  6. SELECT * FROM `degrees` WHERE `level` = 'magistrale';
  7. SELECT COUNT(*) FROM `departments` WHERE 1;
  8. SELECT * FROM `teachers` WHERE `phone`;

Query sql GRUP BY

  1. SELECT COUNT(`id`), YEAR(`enrolment_date`) FROM `students` GROUP BY YEAR(`enrolment_date`);
  2. SELECT COUNT(`id`), `office_address` FROM `teachers` GROUP BY `office_address`;
  3. SELECT AVG(`vote`), `exam_id` AS `media_voti` FROM `exam_student` GROUP BY `exam_id`;
  4. SELECT COUNT(`id`), `department_id` FROM `degrees` GROUP BY `department_id`;

  4. SELECT s.surname, s.name, d.name, dp.name FROM students s INNER JOIN degrees d ON s.degree_id = d.id JOIN departments dp ON dp.id = d.department_id ORDER BY s.surname ASC;
